The Sun Automation

PYTHON 3.10+

Automates the procedural rotation and tilt for the solar surface, corona, and prominence layers.

sun_automation.py
COPIED!
import bpy 
import math 

# --- Settings --- 
TOTAL_FRAMES = 500 
FPS = 24 

# Keywords to find all sun-related layers
KEYWORDS = ["The Sun", "Solar", "Corona", "Prominence", "Star_Surface"] 

scene = bpy.context.scene 
scene.frame_start = 1 
scene.frame_end = TOTAL_FRAMES 
scene.render.fps = FPS 

updated_count = 0 

for obj in bpy.data.objects: 
    if any(key.lower() in obj.name.lower() for key in KEYWORDS): 
        obj.animation_data_clear() 
        obj.rotation_mode = 'XYZ' 
        obj.rotation_euler.x = math.radians(7.25) 
        obj.rotation_euler.z = 0 
        obj.keyframe_insert(data_path="rotation_euler", index=2, frame=1) 
        obj.rotation_euler.z = math.radians(360) 
        obj.keyframe_insert(data_path="rotation_euler", index=2, frame=TOTAL_FRAMES + 1) 

        if obj.animation_data and obj.animation_data.action: 
            action = obj.animation_data.action 
            for layer in action.layers: 
                for strip in layer.strips: 
                    for channelbag in strip.channelbags: 
                        for fcurve in channelbag.fcurves: 
                            for kp in fcurve.keyframe_points: 
                                kp.interpolation = 'LINEAR' 
        updated_count += 1 

if updated_count == 0: 
    print("❌ No sun-related objects found.") 
else: 
    scene.frame_set(1) 
    print(f"✅ Success! Animated {updated_count} sun parts.")

Mercury Automation

PYTHON 3.10+

Implements realistic sidereal day rotation and axial tilt for Mercury based on orbital physics.

mercury_automation.py
COPIED!
import bpy 
import math 

# --- USER SETTINGS --- 
MERCURY_OBJECT_NAME  = "Mercury" 
CAMERA_DISTANCE      = 6.0 
EARTH_DAYS_PER_FRAME = 0.244 
SIDEREAL_DAY_EARTH_DAYS = 58.646 
AXIAL_TILT_DEG          = 0.034 

def center_camera_on_mercury(): 
    camera = bpy.data.objects.get("Camera") 
    if not camera: return 
    camera.location = (0.0, -CAMERA_DISTANCE, 0.0) 
    camera.rotation_euler = (1.5707963, 0.0, 0.0) 

def center_and_spin_mercury(): 
    fps              = bpy.context.scene.render.fps 
    total_frames     = fps * 10 
    mercury = bpy.data.objects.get(MERCURY_OBJECT_NAME) 
    if not mercury: return 

    mercury.location = (0.0, 0.0, 0.0) 
    mercury.animation_data_clear() 

    for frame in range(1, total_frames + 1): 
        bpy.context.scene.frame_set(frame) 
        earth_days = (frame - 1) * EARTH_DAYS_PER_FRAME 
        spin_angle = (earth_days / SIDEREAL_DAY_EARTH_DAYS) * 2 * math.pi 
        mercury.rotation_euler = (math.radians(AXIAL_TILT_DEG), 0.0, spin_angle) 
        mercury.keyframe_insert(data_path="location", frame=frame) 
        mercury.keyframe_insert(data_path="rotation_euler", frame=frame) 

    # Linear Interpolation
    if mercury.animation_data and mercury.animation_data.action: 
        action = mercury.animation_data.action 
        for fcurve in action.fcurves: 
            for kp in fcurve.keyframe_points: 
                kp.interpolation = 'LINEAR' 

center_camera_on_mercury() 
center_and_spin_mercury()

Venus Automation

PYTHON 3.10+

Handles the retrograde rotation and atmospheric layer movement for the Venusian environment.

venus_automation.py
COPIED!
import bpy 
import math 

# --- Venus Settings --- 
VENUS_BODY = "Venus" 
VENUS_CLOUDS = "Venus_Atmosphere" 
SIDEREAL_DAY = 243.022  # Earth days (Retrograde) 
AXIAL_TILT = 177.36     # Degrees 

def apply_venus_rotation(): 
    scene = bpy.context.scene 
    fps = scene.render.fps 
    total_frames = scene.frame_end 
    
    venus = bpy.data.objects.get(VENUS_BODY) 
    clouds = bpy.data.objects.get(VENUS_CLOUDS) 
    
    if not venus: return 

    venus.animation_data_clear() 
    venus.rotation_mode = 'XYZ' 
    
    for frame in range(scene.frame_start, total_frames + 1): 
        scene.frame_set(frame) 
        # Negative sign for retrograde rotation
        spin = -((frame / fps) / (SIDEREAL_DAY * 86400)) * 2 * math.pi 
        venus.rotation_euler = (math.radians(AXIAL_TILT), 0, spin) 
        venus.keyframe_insert(data_path="rotation_euler", frame=frame) 
        
        if clouds: 
            # Clouds rotate slightly faster than the surface
            cloud_spin = spin * 1.2  
            clouds.rotation_euler = (math.radians(AXIAL_TILT), 0, cloud_spin) 
            clouds.keyframe_insert(data_path="rotation_euler", frame=frame) 

apply_venus_rotation()

Earth Automation

PYTHON 3.10+

Synchronizes cloud layer rotation with surface rotation and manages moon orbital constraints.

earth_automation.py
COPIED!
import bpy 
import math 

# --- Earth System Settings --- 
EARTH = "Earth" 
CLOUDS = "Earth_Clouds" 
MOON = "Moon" 
TILT = 23.44 
DAY_LENGTH = 0.9972  # Sidereal day in Earth days 

def setup_earth_system(): 
    scene = bpy.context.scene 
    obj = bpy.data.objects.get(EARTH) 
    cloud_obj = bpy.data.objects.get(CLOUDS) 
    
    if not obj: return 

    # Surface Rotation 
    obj.rotation_mode = 'XYZ' 
    obj.rotation_euler.x = math.radians(TILT) 
    
    # Add driver for constant rotation 
    driver = obj.driver_add("rotation_euler", 2).driver 
    driver.type = 'SCRIPTED' 
    # Rotation formula: (2*pi / (seconds_in_day)) * time 
    driver.expression = f"(2 * pi / ({DAY_LENGTH} * 86400)) * frame / {scene.render.fps}" 
    
    if cloud_obj: 
        cloud_obj.rotation_mode = 'XYZ' 
        cloud_obj.rotation_euler.x = math.radians(TILT) 
        c_driver = cloud_obj.driver_add("rotation_euler", 2).driver 
        c_driver.type = 'SCRIPTED' 
        # Clouds rotate 10% faster for dynamic effect 
        c_driver.expression = f"(2 * pi / ({DAY_LENGTH} * 86400)) * 1.1 * frame / {scene.render.fps}" 

setup_earth_system()

import bpy 
import math 

mars = bpy.data.objects.get("Mars") 
moon1 = bpy.data.objects.get("Moon")      
moon2 = bpy.data.objects.get("Moon.001")  

if not mars: 
    print("ERROR: Mars not found.") 
else: 
    mars.location = (0, 0, 0) 
    mars.scale = (1.0, 1.0, 1.0) 

    if moon1: 
        moon1.scale = (0.14, 0.14, 0.14) 
    if moon2: 
        moon2.scale = (0.09, 0.09, 0.09) 

    bpy.ops.object.select_all(action='DESELECT') 
    for obj in [mars, moon1, moon2]: 
        if obj: 
            obj.animation_data_clear() 

    for obj in list(bpy.data.objects): 
        if obj.type == 'LIGHT': 
            bpy.data.objects.remove(obj, do_unlink=True) 

    sun_data = bpy.data.lights.new(name="Sun", type='SUN') 
    sun_data.energy = 4.0 
    sun_obj = bpy.data.objects.new("Sun", sun_data) 
    bpy.context.collection.objects.link(sun_obj) 
    sun_obj.location = (0, -100, 40) 
    sun_obj.rotation_euler = (math.radians(35), 0, 0) 

    if moon1: 
        empty1 = bpy.data.objects.get("Moon1_Orbit") 
        if not empty1: 
            bpy.ops.object.empty_add(type='PLAIN_AXES', location=(0, 0, 0)) 
            empty1 = bpy.context.active_object 
            empty1.name = "Moon1_Orbit" 
        
        moon1.parent = empty1 
        moon1.location = (3.0, 0, 0.4)   

    if moon2: 
        empty2 = bpy.data.objects.get("Moon2_Orbit") 
        if not empty2: 
            bpy.ops.object.empty_add(type='PLAIN_AXES', location=(0, 0, 0)) 
            empty2 = bpy.context.active_object 
            empty2.name = "Moon2_Orbit" 
        
        moon2.parent = empty2 
        moon2.location = (5.2, 0, -0.4) 

    scene = bpy.context.scene 
    scene.render.fps = 24 
    scene.frame_start = 1 
    scene.frame_end = 240 

    mars.rotation_euler = (math.radians(25.2), 0, 0) 

    for f in range(1, 241, 8): 
        angle = (f - 1) * 360 / 300 
        mars.rotation_euler.z = math.radians(angle) 
        mars.keyframe_insert(data_path="rotation_euler", index=2, frame=f) 

    if moon1 and empty1: 
        empty1.rotation_euler = (0, 0, 0) 
        for f in range(1, 241, 6): 
            angle = (f - 1) * 360 / 180          
            empty1.rotation_euler.z = math.radians(angle) 
            empty1.keyframe_insert(data_path="rotation_euler", index=2, frame=f) 

            moon1.rotation_euler.z = math.radians(angle * 3) 
            moon1.keyframe_insert(data_path="rotation_euler", index=2, frame=f) 

    if moon2 and empty2: 
        empty2.rotation_euler = (0, 0, 0) 
        for f in range(1, 241, 8): 
            angle = (f - 1) * 360 / 320          
            empty2.rotation_euler.z = math.radians(angle) 
            empty2.keyframe_insert(data_path="rotation_euler", index=2, frame=f) 

            moon2.rotation_euler.z = math.radians(angle * 2) 
            moon2.keyframe_insert(data_path="rotation_euler", index=2, frame=f) 

    scene.frame_current = 1 

    print("✅ Stable orbit method applied!")

Jupiter Automation

PYTHON 3.10+

Manages complex differential rotation for the Great Red Spot and Jovian bands.

jupiter_automation.py

import math 
import os 
from pathlib import Path 

import bpy 
from mathutils import Matrix 

JUPITER_NAME = "Jupiter" 
MOON_NAMES = ["Moon", "Moon.001", "Moon.002", "Moon.003"] 
MOON_LABELS = ["Io", "Europa", "Ganymede", "Callisto"] 

FPS = 30 
VIDEO_SECONDS = 10 
FRAME_START = 1 
FRAME_END = FPS * VIDEO_SECONDS 

JUPITER_DIAMETER = 10.0 
MOON_DIAMETERS = { 
    "Io": 1.0,        
    "Europa": 0.6,    
    "Ganymede": 1.8,  
    "Callisto": 1.3,  
} 

ORBIT_DISTANCES = [10.0, 14.0, 19.0, 25.0] 
START_ANGLES_DEG = [205.0, 245.0, 295.0, 335.0] 

JUPITER_SPIN_TURNS = 0.70 
MOON_ORBIT_TURNS = { 
    "Io": 0.90, 
    "Europa": 0.48, 
    "Ganymede": 0.25, 
    "Callisto": 0.12, 
} 

JUPITER_AXIAL_TILT_DEG = 3.13 
AUTO_PREFIX = "AUTO_JUPITER_ONLY_" 


def get_object(name): 
    obj = bpy.data.objects.get(name) 
    if obj is None: 
        raise RuntimeError(f"Missing required object named '{name}'.") 
    return obj 


def clear_previous_setup(): 
    for obj in list(bpy.data.objects): 
        if obj.name.startswith(AUTO_PREFIX): 
            bpy.data.objects.remove(obj, do_unlink=True) 


def clear_object_animation(obj): 
    if obj.animation_data: 
        obj.animation_data_clear() 


def set_keyframes_linear(fcurve): 
    for keyframe in fcurve.keyframe_points: 
        keyframe.interpolation = "LINEAR" 


def iter_layered_action_fcurves(action): 
    for layer in getattr(action, "layers", []): 
        for strip in getattr(layer, "strips", []): 
            for channelbag in getattr(strip, "channelbags", []): 
                for fcurve in getattr(channelbag, "fcurves", []): 
                    yield fcurve 


def set_linear_animation(obj, data_path="rotation_euler", indices=(0, 1, 2)): 
    if not obj.animation_data or not obj.animation_data.action: 
        return 

    action = obj.animation_data.action 

    try: 
        legacy_fcurves = action.fcurves 
    except AttributeError: 
        legacy_fcurves = [] 

    for fcurve in legacy_fcurves: 
        if fcurve.data_path == data_path and fcurve.array_index in indices: 
            set_keyframes_linear(fcurve) 

    for fcurve in iter_layered_action_fcurves(action): 
        if fcurve.data_path == data_path and fcurve.array_index in indices: 
            set_keyframes_linear(fcurve) 


def animate_z_rotation(obj, turns, start_angle=0.0): 
    obj.rotation_mode = "XYZ" 
    obj.rotation_euler = (obj.rotation_euler.x, obj.rotation_euler.y, start_angle) 
    obj.keyframe_insert(data_path="rotation_euler", frame=FRAME_START) 
    obj.rotation_euler[2] = start_angle + (math.tau * turns) 
    obj.keyframe_insert(data_path="rotation_euler", frame=FRAME_END) 
    set_linear_animation(obj) 


def force_exact_diameter(obj, target_diameter): 
    obj.constraints.clear() 
    obj.delta_scale = (1.0, 1.0, 1.0) 
    bpy.context.view_layer.update() 
    
    current_scale_x = obj.scale.x if abs(obj.scale.x) > 0.0001 else 1.0 
    
    raw_width = obj.dimensions.x / current_scale_x 
    if raw_width <= 0.001: 
        raw_width = 1.0 
        
    exact_scale = target_diameter / raw_width 
    obj.scale = (exact_scale, exact_scale, exact_scale) 
    bpy.context.view_layer.update() 


def setup_world_background(): 
    scene = bpy.context.scene 
    if not scene.world: 
        scene.world = bpy.data.worlds.new("World") 
    
    world = scene.world 
    world.use_nodes = True 
    nodes = world.node_tree.nodes 
    links = world.node_tree.links 
    nodes.clear() 

    node_bg = nodes.new(type='ShaderNodeBackground') 
    node_out = nodes.new(type='ShaderNodeOutputWorld') 
    node_tex = nodes.new(type='ShaderNodeTexEnvironment') 

    star_img_path = r"D:\MultimediaPortfolio\images\2k_stars.jpg" 
    if os.path.exists(star_img_path): 
        node_tex.image = bpy.data.images.load(star_img_path) 
        links.new(node_tex.outputs['Color'], node_bg.inputs['Color']) 
        links.new(node_bg.outputs['Background'], node_out.inputs['Surface']) 
        node_bg.inputs['Strength'].default_value = 1.0 


def setup_exhibition_lighting(): 
    for obj in bpy.data.objects: 
        if obj.type == 'LIGHT': 
            bpy.data.objects.remove(obj, do_unlink=True) 

    bpy.ops.object.light_add(type='SUN', align='WORLD') 
    key_light = bpy.context.active_object 
    key_light.name = "Key_Sun" 
    key_light.data.energy = 3.0 
    key_light.rotation_euler = (math.radians(60), math.radians(20), math.radians(45)) 

    bpy.ops.object.light_add(type='SUN', align='WORLD') 
    fill_light = bpy.context.active_object 
    fill_light.name = "Fill_Sun" 
    fill_light.data.energy = 0.6 
    fill_light.rotation_euler = (math.radians(-60), math.radians(200), math.radians(-45)) 


def setup_jupiter_animation(): 
    jupiter = get_object(JUPITER_NAME) 
    moons = [get_object(name) for name in MOON_NAMES] 

    for obj in [jupiter] + moons: 
        obj.parent = None 
        obj.matrix_parent_inverse.identity() 
        obj.location = (0.0, 0.0, 0.0) 
        obj.rotation_euler = (0.0, 0.0, 0.0) 

    clear_previous_setup() 

    scene = bpy.context.scene 
    scene.frame_start = FRAME_START 
    scene.frame_end = FRAME_END 
    scene.render.fps = FPS 

    clear_object_animation(jupiter) 
    for moon in moons: 
        clear_object_animation(moon) 

    force_exact_diameter(jupiter, JUPITER_DIAMETER) 
    for moon, label in zip(moons, MOON_LABELS): 
        force_exact_diameter(moon, MOON_DIAMETERS[label]) 

    center = jupiter.location.copy() 

    jupiter_axis = bpy.data.objects.new(AUTO_PREFIX + "Jupiter_Axis", None) 
    jupiter_axis.empty_display_type = "PLAIN_AXES" 
    jupiter_axis.empty_display_size = JUPITER_DIAMETER * 0.7 
    jupiter_axis.location = center 
    jupiter_axis.rotation_euler = (math.radians(JUPITER_AXIAL_TILT_DEG), 0.0, 0.0) 
    bpy.context.scene.collection.objects.link(jupiter_axis) 

    world_matrix = jupiter.matrix_world.copy() 
    jupiter.parent = jupiter_axis 
    jupiter.matrix_parent_inverse = jupiter_axis.matrix_world.inverted() 
    jupiter.matrix_world = world_matrix 

    animate_z_rotation(jupiter, JUPITER_SPIN_TURNS) 

    moon_orbit_plane = bpy.data.objects.new(AUTO_PREFIX + "Moon_Orbit_Plane", None) 
    moon_orbit_plane.empty_display_type = "PLAIN_AXES" 
    moon_orbit_plane.empty_display_size = JUPITER_DIAMETER * 0.9 
    moon_orbit_plane.location = center 
    moon_orbit_plane.rotation_euler = jupiter_axis.rotation_euler 
    bpy.context.scene.collection.objects.link(moon_orbit_plane) 

    for index, (moon, label) in enumerate(zip(moons, MOON_LABELS)): 
        orbit = bpy.data.objects.new(AUTO_PREFIX + f"{label}_Orbit", None) 
        orbit.empty_display_type = "PLAIN_AXES" 
        orbit.empty_display_size = 2.0 
        orbit.parent = moon_orbit_plane 
        orbit.location = (0.0, 0.0, 0.0) 
        orbit.rotation_euler = (0.0, 0.0, 0.0) 
        bpy.context.scene.collection.objects.link(orbit) 

        moon.parent = orbit 
        moon.matrix_parent_inverse = Matrix.Identity(4) 
        moon.location = (ORBIT_DISTANCES[index], 0.0, 0.0) 

        start_angle = math.radians(START_ANGLES_DEG[index]) 
        animate_z_rotation(orbit, MOON_ORBIT_TURNS[label], start_angle) 
        animate_z_rotation(moon, 0.0) 

setup_world_background() 
setup_exhibition_lighting() 
setup_jupiter_animation() 

print("SLEDGEHAMMER FIX APPLIED: Moon sizes forced perfectly!")
